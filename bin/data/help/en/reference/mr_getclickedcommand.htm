getClickedCommand


function getClickedCommand: command;

Returns the last command that has been clicked.

  var exitCmd, clicked: command;
  begin
    exitCmd := createCommand('Exit', CM_EXIT, 1);
    addCommand(exitCmd);
    repeat
      clicked := getClickedCommand;
    until clicked <> emptyCommand;
  end.
